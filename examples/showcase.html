Basic Setup

// Initialize TypeStorm engine
import { TextEngine } from './src/core/TextEngine.js';

const canvas = document.getElementById('canvas');
const engine = new TextEngine({
  canvas,
  width: 1080,
  height: 1920,
  fps: 60
});

// Create animation
const animation = await engine.applyPreset(
  "HELLO WORLD", 
  "neon", 
  { 
    layoutType: "spiral",
    layoutOptions: { spacing: 80 }
  }
);

// Play animation
engine.play();

Social Media Export

// Export for TikTok (1080x1920, optimized)
const tiktokBlob = await engine.exportForTikTok({
  onProgress: (progress) => {
    console.log(`Export: ${progress.percentage}%`);
  },
  onComplete: (blob) => {
    // Download or upload to TikTok
    downloadBlob(blob, 'tiktok-video.mp4');
  }
});

// Export for Instagram Stories
const instagramBlob = await engine.exportForInstagramStories();

// Export for Twitter
const twitterBlob = await engine.exportForTwitter();

Custom Quality Settings

// High-quality export for presentations
const highQualityBlob = await engine.exportAnimation({
  format: 'mp4',
  quality: 'ultra',
  fps: 60,
  duration: 10,
  width: 1920,
  height: 1080,
  enableHardwareAcceleration: true
});

// Quick preview export
const previewBlob = await engine.exportWithQuality('low');
